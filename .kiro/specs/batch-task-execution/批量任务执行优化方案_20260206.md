# 批量任务执行优化方案

## 📋 背景与问题

### 当前问题

1. **系统提示词开销大**：每次 LLM 请求都携带完整的系统提示词（可能 5000+ tokens）
2. **任务粒度过细**：任务树分解非常细腻，导致大量小任务（每个任务可能只需要 500 字输出）
3. **请求次数过多**：每个小任务单独请求 LLM，导致请求次数激增
4. **LLM 能力未充分利用**：LLM 单次可输出 3000-6000 字，但每次只完成一个小任务

### 成本分析

**示例场景**：生成 10000 字的文章

**当前方案**：
- 分解为 5 个子任务（每个 2000 字）
- 每个任务单独请求 LLM
- 系统提示词：5000 tokens × 5 = 25000 tokens
- 输出：2000 字 × 5 = 10000 字 ≈ 13000 tokens
- **总计**：38000 tokens

**优化后方案**：
- 批量执行 2-3 个任务
- 系统提示词：5000 tokens × 2 = 10000 tokens
- 输出：5000 字 × 2 = 10000 字 ≈ 13000 tokens
- **总计**：23000 tokens
- **节省**：39% tokens

## 🎯 优化目标

1. **批量创建任务**：一次性创建多个相关任务
2. **批量执行任务**：一次 LLM 请求完成多个小任务
3. **智能分组**：根据任务特征自动分组
4. **保持灵活性**：支持单任务执行和批量执行两种模式

#### 2. TaskGrouper（任务分组器）

**功能**：将任务智能分组，决定哪些任务可以批量执行

**分组策略**：
1. **相似度分组**：相似的任务（如连续生成文章的不同段落）
2. **大小分组**：小任务（< 1000 tokens）优先批量执行
3. **依赖关系分组**：无依赖关系的任务可以批量执行

**分组规则**：
- 单个批次最多 3-5 个任务
- 单个批次预估输出 < 6000 tokens
- 批次内任务必须无依赖关系

#### 3. BatchExecutor（批量执行器）

**功能**：将多个任务合并为一次 LLM 请求

**执行流程**：
1. 合并多个任务的 prompt
2. 添加输出格式要求（使用分隔符）
3. 单次 LLM 请求
4. 解析输出并拆分到各个任务
5. 保存结果

**Prompt 模板**：
```
你需要完成以下 {N} 个任务，请按顺序输出结果。

每个任务的输出使用以下格式：

=== TASK {task_id} START ===
{任务输出内容}
=== TASK {task_id} END ===

---

任务 1：{task_1_summary}
{task_1_prompt}

任务 2：{task_2_summary}
{task_2_prompt}

...

请开始输出：
```

### 2. TaskGrouper 实现

```typescript
export class TaskGrouper {
  /**
   * 将任务分组
   */
  groupTasks(tasks: SubTask[], options: GroupingOptions): TaskBatch[] {
    const batches: TaskBatch[] = [];
    const ungrouped = [...tasks];
    
    while (ungrouped.length > 0) {
      const batch = this.createBatch(ungrouped, options);
      if (batch.taskIds.length > 0) {
        batches.push(batch);
        // 移除已分组的任务
        ungrouped.splice(0, batch.taskIds.length);
      } else {
        break;
      }
    }
    
    return batches;
  }
  
  /**
   * 创建一个批次
   */
  private createBatch(tasks: SubTask[], options: GroupingOptions): TaskBatch {
    const batch: TaskBatch = {
      id: crypto.randomUUID(),
      taskIds: [],
      status: "pending",
      createdAt: Date.now(),
      estimatedTokens: 0,
    };
    
    for (const task of tasks) {
      // 检查是否可以加入批次
      if (this.canAddToBatch(batch, task, options)) {
        batch.taskIds.push(task.id);
        batch.estimatedTokens += task.metadata?.estimatedTokens || 1000;
      }
      
      // 检查批次是否已满
      if (batch.taskIds.length >= options.maxTasksPerBatch) {
        break;
      }
      if (batch.estimatedTokens >= options.maxTokensPerBatch) {
        break;
      }
    }
    
    return batch;
  }
  
  /**
   * 检查任务是否可以加入批次
   */
  private canAddToBatch(batch: TaskBatch, task: SubTask, options: GroupingOptions): boolean {
    // 1. 检查任务是否允许批量执行
    if (task.metadata?.canBatch === false) {
      return false;
    }
    
    // 2. 检查批次大小限制
    if (batch.taskIds.length >= options.maxTasksPerBatch) {
      return false;
    }
    
    // 3. 检查 tokens 限制
    const newEstimatedTokens = batch.estimatedTokens + (task.metadata?.estimatedTokens || 1000);
    if (newEstimatedTokens > options.maxTokensPerBatch) {
      return false;
    }
    
    // 4. 检查依赖关系
    if (task.dependencies && task.dependencies.length > 0) {
      // 有依赖关系的任务不能批量执行
      return false;
    }
    
    // 5. 检查复杂度
    if (task.metadata?.complexity === "high") {
      // 高复杂度任务单独执行
      return false;
    }
    
    return true;
  }
}

interface GroupingOptions {
  maxTasksPerBatch: number;      // 单个批次最多任务数（默认 3）
  maxTokensPerBatch: number;     // 单个批次最多 tokens（默认 6000）
  batchMode: "auto" | "force" | "disable";
}
```

### 3. BatchExecutor 实现

```typescript
export class BatchExecutor {
  /**
   * 批量执行任务
   */
  async executeBatch(
    taskTree: TaskTree,
    batch: TaskBatch,
    executor: (prompt: string) => Promise<string>
  ): Promise<void> {
    try {
      // 1. 获取批次中的所有任务
      const tasks = batch.taskIds.map(id => 
        taskTree.subTasks.find(t => t.id === id)
      ).filter(Boolean) as SubTask[];
      
      // 2. 合并 prompt
      const mergedPrompt = this.mergePrompts(tasks);
      batch.mergedPrompt = mergedPrompt;
      
      // 3. 执行 LLM 请求
      const mergedOutput = await executor(mergedPrompt);
      batch.mergedOutput = mergedOutput;
      
      // 4. 拆分输出
      const outputs = this.splitOutput(mergedOutput, tasks);
      
      // 5. 保存结果到各个任务
      for (let i = 0; i < tasks.length; i++) {
        const task = tasks[i];
        const output = outputs[i];
        
        task.output = output;
        task.completedAt = Date.now();
        task.status = "completed";
        
        // 更新元数据
        if (task.metadata) {
          task.metadata.actualTokens = this.estimateTokens(output);
          task.metadata.batchId = batch.id;
          task.metadata.batchIndex = i;
        }
      }
      
      // 6. 更新批次状态
      batch.status = "completed";
      batch.completedAt = Date.now();
      batch.actualTokens = this.estimateTokens(mergedOutput);
      
      console.log(`[BatchExecutor] ✅ Batch ${batch.id} completed: ${tasks.length} tasks`);
    } catch (err) {
      batch.status = "failed";
      throw err;
    }
  }
  
  /**
   * 合并多个任务的 prompt
   */
  private mergePrompts(tasks: SubTask[]): string {
    let merged = `你需要完成以下 ${tasks.length} 个任务，请按顺序输出结果。\n\n`;
    merged += `每个任务的输出使用以下格式：\n\n`;
    merged += `=== TASK {task_id} START ===\n`;
    merged += `{任务输出内容}\n`;
    merged += `=== TASK {task_id} END ===\n\n`;
    merged += `---\n\n`;
    
    for (let i = 0; i < tasks.length; i++) {
      const task = tasks[i];
      merged += `任务 ${i + 1}：${task.summary}\n`;
      merged += `${task.prompt}\n\n`;
    }
    
    merged += `请开始输出：\n`;
    
    return merged;
  }
  
  /**
   * 拆分合并后的输出
   */
  private splitOutput(mergedOutput: string, tasks: SubTask[]): string[] {
    const outputs: string[] = [];
    
    for (const task of tasks) {
      const startMarker = `=== TASK ${task.id} START ===`;
      const endMarker = `=== TASK ${task.id} END ===`;
      
      const startIndex = mergedOutput.indexOf(startMarker);
      const endIndex = mergedOutput.indexOf(endMarker);
      
      if (startIndex !== -1 && endIndex !== -1) {
        const output = mergedOutput.substring(
          startIndex + startMarker.length,
          endIndex
        ).trim();
        outputs.push(output);
      } else {
        // 如果找不到标记，尝试按顺序拆分
        console.warn(`[BatchExecutor] ⚠️ Cannot find markers for task ${task.id}, using fallback`);
        outputs.push(this.fallbackSplit(mergedOutput, tasks.indexOf(task), tasks.length));
      }
    }
    
    return outputs;
  }
  
  /**
   * 后备拆分方法（如果找不到标记）
   */
  private fallbackSplit(mergedOutput: string, index: number, total: number): string {
    // 简单地按长度平均拆分
    const chunkSize = Math.floor(mergedOutput.length / total);
    const start = index * chunkSize;
    const end = (index === total - 1) ? mergedOutput.length : (index + 1) * chunkSize;
    return mergedOutput.substring(start, end).trim();
  }
  
  /**
   * 估算 tokens 数量
   */
  private estimateTokens(text: string): number {
    // 简单估算：中文 1 字 ≈ 1.5 tokens，英文 1 词 ≈ 1.3 tokens
    const chineseChars = (text.match(/[\u4e00-\u9fa5]/g) || []).length;
    const englishWords = (text.match(/[a-zA-Z]+/g) || []).length;
    return Math.ceil(chineseChars * 1.5 + englishWords * 1.3);
  }
}
```

```typescript
/**
 * 智能预估任务输出 tokens

function estimateTaskTokens(task: SubTask): number {
  // 1. 从 prompt 中提取字数要求
  const wordCountMatch = task.prompt.match(/(\d+)\s*字/);
  if (wordCountMatch) {
    const wordCount = parseInt(wordCountMatch[1]);
    return Math.ceil(wordCount * 1.5); // 中文 1 字 ≈ 1.5 tokens
  }
  
  // 2. 从 summary 中提取字数要求
  const summaryMatch = task.summary.match(/(\d+)\s*字/);
  if (summaryMatch) {
    const wordCount = parseInt(summaryMatch[1]);
    return Math.ceil(wordCount * 1.5);
  }
  
  // 3. 根据复杂度估算
  if (task.metadata?.complexity === "high") {
    return 3000;
  } else if (task.metadata?.complexity === "medium") {
    return 1500;
  } else {
    return 1000;
  }
}
```

### 2. 动态调整

```typescript
/**
 * 根据实际执行情况动态调整分组策略
 */
class AdaptiveGrouper extends TaskGrouper {
  private history: Array<{
    batchSize: number;
    estimatedTokens: number;
    actualTokens: number;
    duration: number;
  }> = [];
  
  /**
   * 记录执行历史
   */
  recordExecution(batch: TaskBatch, duration: number): void {
    this.history.push({
      batchSize: batch.taskIds.length,
      estimatedTokens: batch.estimatedTokens,
      actualTokens: batch.actualTokens || 0,
      duration,
    });
    
    // 只保留最近 20 次记录
    if (this.history.length > 20) {
      this.history.shift();
    }
  }
  
  /**
   * 根据历史调整分组参数
   */
  getAdaptiveOptions(): GroupingOptions {
    if (this.history.length < 5) {
      // 历史记录不足，使用默认值
      return {
        maxTasksPerBatch: 3,
        maxTokensPerBatch: 6000,
        batchMode: "auto",
      };
    }
    
    // 计算平均误差
    const avgError = this.history.reduce((sum, record) => {
      return sum + Math.abs(record.actualTokens - record.estimatedTokens) / record.estimatedTokens;
    }, 0) / this.history.length;
    
    // 如果误差较大，减小批次大小
    if (avgError > 0.3) {
      return {
        maxTasksPerBatch: 2,
        maxTokensPerBatch: 4000,
        batchMode: "auto",
      };
    }
    
    // 如果误差较小，可以增大批次大小
    if (avgError < 0.1) {
      return {
        maxTasksPerBatch: 5,
        maxTokensPerBatch: 8000,
        batchMode: "auto",
      };
    }
    
    // 默认值
    return {
      maxTasksPerBatch: 3,
      maxTokensPerBatch: 6000,
      batchMode: "auto",
    };
  }
}
```

```typescript
/**
 * 如果输出拆分失败，回退到单任务执行

async function executeBatchWithFallback(
  taskTree: TaskTree,
  batch: TaskBatch,
  executor: (prompt: string) => Promise<string>
): Promise<void> {
  try {
    // 尝试批量执行
    await batchExecutor.executeBatch(taskTree, batch, executor);
  } catch (err) {
    console.warn(`[BatchExecutor] ⚠️ Batch execution failed, falling back to single task execution`);
    
    // 回退到单任务执行
    const tasks = batch.taskIds.map(id => 
      taskTree.subTasks.find(t => t.id === id)
    ).filter(Boolean) as SubTask[];
    
    for (const task of tasks) {
      try {
        const output = await executor(task.prompt);
        task.output = output;
        task.completedAt = Date.now();
        task.status = "completed";
      } catch (taskErr) {
        task.error = String(taskErr);
        task.status = "failed";
      }
    }
  }
}
```
